## ÉTAPE 5 — `src/app/core/services/auth.service.ts`

```typescript
import { Injectable } from '@angular/core';

@Injectable({ providedIn: 'root' })
export class AuthService {

  // Le token JWT vient de ton service d'authentification principal
  // (user-service, gateway, etc.)
  // Ce service lit simplement le token stocké

  getToken(): string | null {
    return localStorage.getItem('access_token');
  }

  setToken(token: string): void {
    localStorage.setItem('access_token', token);
  }

  // Décode manuellement le JWT pour extraire les claims
  // (userId, name, image sont dans le token d'après JwtService.java)
  decodeToken(): any {
    const token = this.getToken();
    if (!token) return null;
    try {
      const payload = token.split('.')[1];
      return JSON.parse(atob(payload));
    } catch {
      return null;
    }
  }

  getUserId(): string | null {
    return this.decodeToken()?.userId ?? null;
  }

  getUserName(): string | null {
    return this.decodeToken()?.name ?? null;
  }

  getUserImage(): string | null {
    return this.decodeToken()?.image ?? null;
  }

  isLoggedIn(): boolean {
    return !!this.getToken();
  }

  logout(): void {
    localStorage.removeItem('access_token');
  }
}
```

## ÉTAPE 6 — `src/app/core/interceptors/jwt.interceptor.ts`

```typescript
import { Injectable } from '@angular/core';
import { HttpRequest, HttpHandler, HttpEvent, HttpInterceptor } from '@angular/common/http';
import { Observable } from 'rxjs';
import { AuthService } from '../services/auth.service';

@Injectable()
export class JwtInterceptor implements HttpInterceptor {

  constructor(private authService: AuthService) {}

  intercept(request: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
    const token = this.authService.getToken();
    if (token) {
      request = request.clone({
        setHeaders: { Authorization: `Bearer ${token}` }
      });
    }
    return next.handle(request);
  }
}
```

## ÉTAPE 7 — `src/app/core/services/chat-api.service.ts`

> Service REST — correspond à `ChatRestController.java`

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { ChatMessage, ChatUser, ContactModel, GroupUser } from '../models/chat.model';

@Injectable({ providedIn: 'root' })
export class ChatApiService {

  // ⚠️ Remplace par l'URL de ton backend Spring Boot
  private readonly API = 'http://localhost:8081/api/chat';

  constructor(private http: HttpClient) {}

  // === USERS ===
  // GET /api/chat/users  → liste des ChatUser
  getUsers(): Observable<ChatUser[]> {
    return this.http.get<ChatUser[]>(`${this.API}/users`);
  }

  // === MESSAGES ===
  // GET /api/chat/messages/{chatRoomId}
  getMessages(chatRoomId: string): Observable<ChatMessage[]> {
    return this.http.get<ChatMessage[]>(`${this.API}/messages/${chatRoomId}`);
  }

  // DELETE /api/chat/messages/{messageId}
  deleteMessage(messageId: string): Observable<void> {
    return this.http.delete<void>(`${this.API}/messages/${messageId}`);
  }

  // === ROOMS ===
  // GET /api/chat/rooms
  getChatRooms(): Observable<any[]> {
    return this.http.get<any[]>(`${this.API}/rooms`);
  }

  // POST /api/chat/rooms/connect  → créer/trouver une room avec un user
  connectRoom(payload: { userId: string; targetUserId: string }): Observable<any> {
    return this.http.post<any>(`${this.API}/rooms/connect`, payload);
  }

  // === CONTACTS ===
  // GET /api/chat/contacts  → ContactModelResponse[]
  getContacts(): Observable<ContactModel[]> {
    return this.http.get<ContactModel[]>(`${this.API}/contacts`);
  }

  // === GROUPS ===
  // GET /api/chat/groups  → GroupUserResponse[]
  getGroups(): Observable<GroupUser[]> {
    return this.http.get<GroupUser[]>(`${this.API}/groups`);
  }
}
```

## ÉTAPE 8 — `src/app/core/services/chat-ws.service.ts`

> Service WebSocket STOMP — correspond à `ChatWebSocketController.java` et `WebSocketConfig.java`

```typescript
import { Injectable, OnDestroy } from '@angular/core';
import { Client, IMessage } from '@stomp/stompjs';
import * as SockJS from 'sockjs-client';
import { Subject, Observable } from 'rxjs';
import { ChatMessage, MessageRequest } from '../models/chat.model';
import { AuthService } from './auth.service';

@Injectable({ providedIn: 'root' })
export class ChatWsService implements OnDestroy {

  // ⚠️ Remplace par l'URL WebSocket de ton backend
  private readonly WS_URL = 'http://localhost:8081/ws';

  private client!: Client;
  private messageSubject = new Subject<ChatMessage>();
  private connected = false;

  constructor(private authService: AuthService) {}

  connect(): void {
    const token = this.authService.getToken();

    this.client = new Client({
      webSocketFactory: () => new SockJS(this.WS_URL),
      connectHeaders: {
        Authorization: `Bearer ${token}`
      },
      reconnectDelay: 5000,
      onConnect: () => {
        this.connected = true;
        console.log('✅ WebSocket connecté');
      },
      onStompError: (frame) => {
        console.error('❌ Erreur STOMP:', frame);
      }
    });

    this.client.activate();
  }

  // Écouter les messages d'une room
  // Correspond à @MessageMapping dans ChatWebSocketController.java
  subscribeToRoom(chatRoomId: string): Observable<ChatMessage> {
    const subject = new Subject<ChatMessage>();

    const waitAndSubscribe = () => {
      if (this.client?.connected) {
        // Le topic STOMP typique : /topic/messages/{chatRoomId}
        this.client.subscribe(`/topic/messages/${chatRoomId}`, (msg: IMessage) => {
          const chatMsg: ChatMessage = JSON.parse(msg.body);
          subject.next(chatMsg);
        });
      } else {
        setTimeout(waitAndSubscribe, 300);
      }
    };
    waitAndSubscribe();

    return subject.asObservable();
  }

  // Envoyer un message via WebSocket
  // Correspond à @MessageMapping('/chat.send') ou similaire
  sendMessage(payload: MessageRequest): void {
    if (this.client?.connected) {
      this.client.publish({
        destination: '/app/chat.send',
        body: JSON.stringify(payload)
      });
    }
  }

  disconnect(): void {
    this.client?.deactivate();
    this.connected = false;
  }

  ngOnDestroy(): void {
    this.disconnect();
  }
}
```

## ÉTAPE 9 — `src/app/chat/chat.component.ts` (version complète avec backend)

```typescript
import { Component, OnInit, OnDestroy, ViewChild, TemplateRef } from '@angular/core';
import { UntypedFormBuilder, Validators, UntypedFormGroup } from '@angular/forms';
import { NgbOffcanvas } from '@ng-bootstrap/ng-bootstrap';
import { DatePipe } from '@angular/common';
import { Lightbox } from 'ngx-lightbox';
import { Subscription } from 'rxjs';

import { ChatUser, GroupUser, ChatMessage, ContactModel, MessageRequest } from '../core/models/chat.model';
import { ChatApiService } from '../core/services/chat-api.service';
import { ChatWsService } from '../core/services/chat-ws.service';
import { AuthService } from '../core/services/auth.service';

@Component({
  selector: 'app-chat',
  templateUrl: './chat.component.html',
  styleUrls: ['./chat.component.scss']
})
export class ChatComponent implements OnInit, OnDestroy {

  chatData: ChatUser[] = [];
  groupData: GroupUser[] = [];
  chatMessagesData: ChatMessage[] = [];
  contactData: ContactModel[] = [];

  formData!: UntypedFormGroup;
  submitted = false;
  isStatus: string = 'online';
  isProfile: string = 'assets/images/users/user-dummy-img.jpg';
  username: string = '';
  isreplyMessage = false;
  emoji = '';
  showEmojiPicker = false;

  currentChatRoomId: string = '';
  currentUserId: string = '';
  currentUserName: string = '';
  currentUserProfile: string = '';

  private wsSub?: Subscription;
  images: { src: string; thumb: string; caption: string }[] = [];

  @ViewChild('scrollRef') scrollRef: any;

  constructor(
    public formBuilder: UntypedFormBuilder,
    private lightbox: Lightbox,
    private offcanvasService: NgbOffcanvas,
    private datePipe: DatePipe,
    private chatApi: ChatApiService,
    private chatWs: ChatWsService,
    private authService: AuthService
  ) {}

  ngOnInit(): void {
    // Récupérer les infos de l'utilisateur connecté depuis le JWT
    this.currentUserId   = this.authService.getUserId() || '';
    this.currentUserName = this.authService.getUserName() || 'Moi';
    this.currentUserProfile = this.authService.getUserImage() || '';

    this.formData = this.formBuilder.group({
      message: ['', [Validators.required]],
    });

    // Connexion WebSocket
    this.chatWs.connect();

    // Charger les données depuis le backend
    this.loadUsers();
    this.loadGroups();
    this.loadContacts();
  }

  ngOnDestroy(): void {
    this.wsSub?.unsubscribe();
    this.chatWs.disconnect();
  }

  loadUsers(): void {
    this.chatApi.getUsers().subscribe({
      next: (users) => this.chatData = users,
      error: (err) => console.error('Erreur chargement users:', err)
    });
  }

  loadGroups(): void {
    this.chatApi.getGroups().subscribe({
      next: (groups) => this.groupData = groups,
      error: (err) => console.error('Erreur chargement groups:', err)
    });
  }

  loadContacts(): void {
    this.chatApi.getContacts().subscribe({
      next: (contacts) => this.contactData = contacts,
      error: (err) => console.error('Erreur chargement contacts:', err)
    });
  }

  loadMessages(chatRoomId: string): void {
    this.chatApi.getMessages(chatRoomId).subscribe({
      next: (messages) => {
        this.chatMessagesData = messages;
        this.onListScroll();
      },
      error: (err) => console.error('Erreur chargement messages:', err)
    });
  }

  chatUsername(name: string, profile: any, status: string): void {
    this.username  = name;
    this.isStatus  = status;
    this.isProfile = profile || 'assets/images/users/user-dummy-img.jpg';

    // Ouvrir/créer la chat room
    this.chatApi.connectRoom({
      userId: this.currentUserId,
      targetUserId: name   // À adapter selon le vrai identifiant backend
    }).subscribe({
      next: (room) => {
        this.currentChatRoomId = room.id;
        this.loadMessages(room.id);
        this.subscribeToRoom(room.id);
      },
      error: (err) => console.error('Erreur connexion room:', err)
    });

    document.querySelector('.user-chat')?.classList.add('user-chat-show');
  }

  private subscribeToRoom(chatRoomId: string): void {
    this.wsSub?.unsubscribe();
    this.wsSub = this.chatWs.subscribeToRoom(chatRoomId).subscribe({
      next: (msg: ChatMessage) => {
        // Ne pas ajouter les messages déjà envoyés par nous
        if (msg.name !== this.currentUserName) {
          this.chatMessagesData.push(msg);
          this.onListScroll();
        }
      }
    });
  }

  get form() {
    return this.formData.controls;
  }

  messageSave(): void {
    const message = this.formData.get('message')!.value;
    if (!message) { this.submitted = true; return; }

    const payload: MessageRequest = {
      chatRoomId: this.currentChatRoomId,
      senderId:   this.currentUserId,
      name:       this.currentUserName,
      profile:    this.currentUserProfile,
      message,
      align:      'right',
    };

    // Gérer reply
    if (this.isreplyMessage) {
      const replyNameEl = document.querySelector('.replyCard .replymessage-block .flex-grow-1 .conversation-name') as HTMLElement;
      const replyMsgEl  = document.querySelector('.replyCard .replymessage-block .flex-grow-1 .mb-0') as HTMLElement;
      payload.align = 'right';
    }

    // Envoyer via WebSocket
    this.chatWs.sendMessage(payload);

    // Affichage immédiat côté émetteur
    this.chatMessagesData.push({
      ...payload,
      time: this.datePipe.transform(new Date(), 'h:mm a') || '',
    });

    this.formData.reset();
    this.emoji = '';
    this.submitted = true;
    this.isreplyMessage = false;
    document.querySelector('.replyCard')?.classList.remove('show');
    this.onListScroll();
  }

  // ===== Méthodes UI (inchangées) =====

  onListScroll(): void {
    if (this.scrollRef !== undefined) {
      setTimeout(() => {
        this.scrollRef.SimpleBar.getScrollElement().scrollTop =
          this.scrollRef.SimpleBar.getScrollElement().scrollHeight;
      }, 500);
    }
  }

  SidebarHide(): void {
    document.querySelector('.user-chat')?.classList.remove('user-chat-show');
  }

  onChatInfoClicked(content: TemplateRef<any>): void {
    this.offcanvasService.open(content, { position: 'end' });
  }

  ContactSearch(): void {
    const input  = document.getElementById('searchContact') as HTMLInputElement;
    const filter = input.value.toUpperCase();
    document.querySelectorAll('.chat-user-list').forEach((ul: any) => {
      Array.from(ul.getElementsByTagName('li')).forEach((li: any) => {
        const txt = li.getElementsByTagName('p')[0]?.innerText || '';
        li.style.display = txt.toUpperCase().includes(filter) ? '' : 'none';
      });
    });
  }

  MessageSearch(): void {
    const input  = document.getElementById('searchMessage') as HTMLInputElement;
    const filter = input.value.toUpperCase();
    const ul     = document.getElementById('users-conversation');
    Array.from(ul?.getElementsByTagName('li') || []).forEach((li: any) => {
      const txt    = li.getElementsByTagName('p')[0]?.innerText || '';
      li.style.display = txt.toUpperCase().includes(filter) ? '' : 'none';
    });
  }

  replyMessage(event: any, align: any): void {
    this.isreplyMessage = true;
    document.querySelector('.replyCard')?.classList.add('show');
    const copyText   = event.target.closest('.chat-list')?.querySelector('.ctext-content')?.innerHTML;
    const nameEl     = document.querySelector('.replyCard .replymessage-block .flex-grow-1 .mb-0') as HTMLElement;
    const replyNameEl = document.querySelector('.replyCard .replymessage-block .flex-grow-1 .conversation-name') as HTMLElement;
    if (nameEl) nameEl.innerHTML = copyText;
    const owner = event.target.closest('.chat-list')?.classList.contains('right')
      ? 'Vous'
      : (document.querySelector('.username') as HTMLElement)?.innerHTML;
    if (replyNameEl) replyNameEl.innerHTML = owner;
  }

  copyMessage(event: any): void {
    navigator.clipboard.writeText(
      event.target.closest('.chat-list')?.querySelector('.ctext-content')?.innerHTML || ''
    );
    const alert = document.getElementById('copyClipBoard') as HTMLElement;
    if (alert) { alert.style.display = 'block'; setTimeout(() => alert.style.display = 'none', 1000); }
  }

  deleteMessage(event: any): void {
    const li = event.target.closest('.chat-list');
    // Optionnel: appel API pour soft-delete
    // this.chatApi.deleteMessage(messageId).subscribe();
    li?.remove();
  }

  deleteAllMessage(event: any): void {
    document.getElementById('users-conversation')
      ?.querySelectorAll('.chat-list')
      .forEach((el: any) => el.remove());
  }

  toggleEmojiPicker(): void { this.showEmojiPicker = !this.showEmojiPicker; }
  addEmoji(event: any): void {
    this.emoji = `${this.emoji}${event.emoji.native}`;
    this.showEmojiPicker = false;
  }
  onFocus(): void { this.showEmojiPicker = false; }
  onBlur():  void {}
  closeReplay(): void { document.querySelector('.replyCard')?.classList.remove('show'); }
  delete(event: any): void { event.target.closest('li')?.remove(); }
  open(index: number): void { this.lightbox.open(this.images, index, {}); }
}
```

## ÉTAPE 14 — CORS dans le backend Spring Boot

Dans ton `SecurityConfig.java` ou une config dédiée, assure-toi d'autoriser Angular :

```java
@Bean
public CorsConfigurationSource corsConfigurationSource() {
    CorsConfiguration config = new CorsConfiguration();
    config.setAllowedOrigins(List.of("http://localhost:4200"));
    config.setAllowedMethods(List.of("GET", "POST", "PUT", "DELETE", "OPTIONS"));
    config.setAllowedHeaders(List.of("*"));
    config.setAllowCredentials(true);

    UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
    source.registerCorsConfiguration("/**", config);
    return source;
}
```

Et dans `WebSocketConfig.java` :

```java
@Override
public void registerStompEndpoints(StompEndpointRegistry registry) {
    registry.addEndpoint("/ws")
            .setAllowedOriginPatterns("http://localhost:4200")
            .withSockJS();
}

@Override
public void configureMessageBroker(MessageBrokerRegistry registry) {
    registry.enableSimpleBroker("/topic", "/queue");
    registry.setApplicationDestinationPrefixes("/app");
}
```

## Points clés du backend détectés

- **JWT** : Le token contient `userId`, `name`, `image`, `email`, `role` — tout est extrait automatiquement par `AuthService.decodeToken()`
- **Message** : Stocké dans MongoDB collection `chat_messages`, avec support reply, images, soft-delete
- **MessageService** : Gère save/getByRoom/delete avec `chatRoomId`
- **Spring Boot 4.0.2** avec Java 17
- **Lombok** : `@Data @Builder @NoArgsConstructor @AllArgsConstructor` sur les entités
